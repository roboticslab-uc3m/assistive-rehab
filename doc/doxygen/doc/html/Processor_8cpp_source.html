110 double Rom_Processor::computeMetric()
111 {
112  double result;
113  Vector v1(3,0.0);
114  Vector ref_dir(3,0.0);
115 
116  //get reference keypoint from skeleton
117  string tag_joint = rom->getTagJoint();
118 
119  if(curr_skeleton[tag_joint]->isUpdated() && curr_skeleton[tag_joint]->getChild(0)->isUpdated())
120  {
121  Vector kp_ref = curr_skeleton[tag_joint]->getPoint();
122 
123  double theta;
124  if(curr_skeleton[tag_joint]->getNumChild())
125  {
126  Vector kp_child = curr_skeleton[tag_joint]->getChild(0)->getPoint();
127 
128  if(rom->getTagPlane() == "coronal")
129  {
130  Vector cor(3,0.0);
131  cor[0]=1.0;
132  plane_normal = cor;
133  }
134  else if(rom->getTagPlane() == "sagittal")
135  {
136  Vector sag(3,0.0);
137  sag[1]=1.0;
138  plane_normal = sag;
139  }
140  else if(rom->getTagPlane() == "transverse")
141  {
142  Vector trans(3,0.0);
143  trans[2]=1.0;
144  plane_normal = trans;
145  }
146 
147  ref_dir = rom->getRefDir();
148 
149  Vector k1=kp_ref;
150  k1.push_back(1.0);
151  Vector k2=kp_child;
152  k2.push_back(1.0);
153  Vector transformed_kp_ref = inv_reference_system*k1;
154  Vector transformed_kp_child = inv_reference_system*k2;
155 
156  v1 = transformed_kp_child.subVector(0,2)-transformed_kp_ref.subVector(0,2);
157  double dist = dot(v1,plane_normal);
158  v1 = v1-dist*plane_normal;
159  double n1 = norm(v1);
160  if(n1 > 0.0)
161  v1 /= n1;
162 
163  double n2 = norm(ref_dir);
164  double dot_p = dot(v1,ref_dir);
165 
166  theta = acos(dot_p/n2);
167  result = theta * (180/M_PI);
168  prev_result = result;
169  }
170  else
171  {
172  yError() << "The keypoint does not have a child ";
173  result = 0.0;
174  }
175  }
176  else
177  {
178  result = prev_result;
179  }
180 
181  return result;
182 
183 }
